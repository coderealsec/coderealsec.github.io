###################

#!/bin/bash --> Bash betiğini burda belirtiyoruz.

counter=1 --> sayaç olarak counter değeri veriyoruz. Counter değerini 1 veriyoruz. Enviroment olarak 1 değerini burada 1 vermemizin sebebi ise değişken mantığı ile düşünebiliriz.

while [ $counter -le 10 ]; do --> while döngüsüne sokarak  counter değerini 1'den 10 kadar saymasını sağlamak için burada yapılacak işlemi belirtiyoruz.
    echo $counter             --> echo $counter ile input değerini ekranda görmesini sağlıyoruz.
    ((counter++))             --> counter++ ile sayacın her adımda artmasını sağlıyoruz.
done                          --> done komutu ile komutu sonlandırıyoruz.
